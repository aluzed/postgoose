     console.trace(err);

    Database config

    Query string
    callback function

    (err, results) => {...}
    Throws:
    ConnectionNotInitialized
    Returns:
    Promise -

    Bluebird Promise

    Example
    postgoose.runOnce({
  "host"     : "localhost",
  "port"     : 5432,
  "database" : "database_name",
  "user"     : "johndoe",
  "password" : "qwerty"
}, "SELECT id, username, email FROM users WHERE active = t", (err, results) => {
   if(err)
     console.trace(err);

   console.log(results);
})

// Or with promises
postgoose..runOnce({
  "host"     : "localhost",
  "port"     : 5432,
  "database" : "database_name",
  "user"     : "johndoe",
  "password" : "qwerty"
}, "SELECT * FROM table").then(results => {
   console.log(results);
